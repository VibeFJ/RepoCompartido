SELECT        US_ID, KE_NUMEMP, ID_SUCURSAL, ID_INMUEBLE, SI_ID, OP_ID, OP_NOMBRE, OP_PADREID, OP_POSICION, OP_VENTANA, OP_DIALOG, MAX(PERMISO_1) AS P01, MAX(PERMISO_2) AS P02, MAX(PERMISO_3) AS P03, 
                         MAX(PERMISO_4) AS P04, MAX(PERMISO_5) AS P05, MAX(PERMISO_6) AS P06, MAX(PERMISO_7) AS P07, MAX(PERMISO_8) AS P08, MAX(PERMISO_9) AS P09, MAX(PERMISO_10) AS P10
FROM            (SELECT        us.US_ID, esu.KE_NUMEMP, esu.ID_SUCURSAL, esu.ID_INMUEBLE, op.SI_ID, op.OP_ID, op.OP_VENTANA, op.OP_DIALOG, peu.PERMISO_1, peu.PERMISO_2, peu.PERMISO_3, peu.PERMISO_4, peu.PERMISO_5, 
                                                    peu.PERMISO_6, peu.PERMISO_7, peu.PERMISO_8, peu.PERMISO_9, peu.PERMISO_10, op.OP_NOMBRE, op.OP_PADREID, op.OP_POSICION
                          FROM            dbo.SEG_USUARIOS AS us INNER JOIN
                                                    dbo.SEG_EMPSUC_USUARIO AS esu ON us.US_ID = esu.US_ID INNER JOIN
                                                    dbo.SEG_PERMISOS_USUARIOS AS peu ON esu.US_ID = peu.US_ID AND esu.KE_NUMEMP = peu.KE_NUMEMP AND esu.ID_SUCURSAL = peu.ID_SUCURSAL AND 
                                                    esu.ID_INMUEBLE = peu.ID_INMUEBLE INNER JOIN
                                                    dbo.SEG_OPCIONES AS op ON peu.SI_ID = op.SI_ID AND peu.OP_ID = op.OP_ID
                          UNION
                          SELECT        us.US_ID, esp.KE_NUMEMP, esp.ID_SUCURSAL, esp.ID_INMUEBLE, pep.SI_ID, pep.OP_ID, op.OP_VENTANA, op.OP_DIALOG, pep.PERMISO_1, pep.PERMISO_2, pep.PERMISO_3, pep.PERMISO_4, 
                                                   pep.PERMISO_5, pep.PERMISO_6, pep.PERMISO_7, pep.PERMISO_8, pep.PERMISO_9, pep.PERMISO_10, op.OP_NOMBRE, op.OP_PADREID, op.OP_POSICION
                          FROM            dbo.SEG_USUARIOS AS us INNER JOIN
                                                   dbo.SEG_PERFIL_USUARIO AS pu ON us.US_ID = pu.US_ID INNER JOIN
                                                   dbo.SEG_EMPSUC_PERFIL AS esp ON pu.PE_ID = esp.PE_ID INNER JOIN
                                                   dbo.SEG_PERMISOS_PERFIL AS pep ON esp.PE_ID = pep.PE_ID AND esp.KE_NUMEMP = pep.KE_NUMEMP AND esp.ID_SUCURSAL = pep.ID_SUCURSAL AND esp.ID_INMUEBLE = pep.ID_INMUEBLE INNER JOIN
                                                   dbo.SEG_OPCIONES AS op ON pep.SI_ID = op.SI_ID AND pep.OP_ID = op.OP_ID
                          UNION
                          SELECT        u.US_ID, pu.KE_NUMEMP, pu.ID_SUCURSAL, pu.ID_INMUEBLE, pp.SI_ID, pp.OP_ID, o.OP_VENTANA, o.OP_DIALOG, pp.PERMISO_1, pp.PERMISO_2, pp.PERMISO_3, pp.PERMISO_4, pp.PERMISO_5, 
                                                   pp.PERMISO_6, pp.PERMISO_7, pp.PERMISO_8, pp.PERMISO_9, pp.PERMISO_10, o.OP_NOMBRE, o.OP_PADREID, o.OP_POSICION
                          FROM            dbo.SEG_USUARIOS AS u INNER JOIN
                                                   dbo.SEG_USUARIO_PERFIL_GENERICO AS pu ON u.US_ID = pu.US_ID INNER JOIN
                                                   dbo.SEG_PERMISOS_PERFIL_GENERICO AS pp ON pu.PE_ID = pp.PE_ID INNER JOIN
                                                   dbo.SEG_OPCIONES AS o ON pp.OP_ID = o.OP_ID) AS tabla
GROUP BY US_ID, KE_NUMEMP, ID_SUCURSAL, ID_INMUEBLE, SI_ID, OP_ID, OP_NOMBRE, OP_PADREID, OP_POSICION, OP_VENTANA, OP_DIALOG
